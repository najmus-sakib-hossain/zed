        &mut self,

    ) -> Self::Result<R>;

        build_view: impl FnOnce(&mut Window, &mut Context<'_, V>) -> V,

    fn focus<V>(&mut self, entity: &Entity<V>) -> Self::Result<()>

}

Expand description

This trait is used for the different visual contexts in GPUI that
require a window to be present.
